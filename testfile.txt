	$(function() {
   
		$('#tag').change(function() {		 
			$('#exerciseDropdown').attr("disabled","disabled");
			var sel_exercise = $(this).val();
			$.getJSON( "http://wod-gen.com/exercises-menu-based-on-tag.php", {
				id_option: sel_exercise
			} )
				
			.done(function( json ) {
				//console.log( "JSON Data: " + json);
				var sess_exercise = ""
				console.log("sess_exericse: " + sess_exercise);
				$('select#exerciseDropdown').empty();
				$('select#exerciseDropdown').append('<option value="all">Any exercise</option>');
				$('select#exerciseDropdown').append('<option value="all">-</option>');
				   
				$.each( json, function( key, value ) {
					//console.log( key + ": " + value );
					if (value == sess_exercise) {
						$('select#exerciseDropdown').append('<option value="'+value+'" selected>'+value+'</option>');
					} else {
						$('select#exerciseDropdown').append('<option value="'+value+'">'+value+'</option>');
					}
				});
				$('#exerciseDropdown').attr("disabled", false);
				$('select#exerciseDropdown').focus();
			})
			.fail(function( jqxhr, textStatus, error ) {
				var err = textStatus + ', ' + error;
				$('select#exerciseDropdown').empty();
				console.log( "Request Failed: " + err);
				console.log( "jqXHR: " + jqxhr);
			});
		}); //END dropdown change event
		
  	  			
	}); //END document.ready
